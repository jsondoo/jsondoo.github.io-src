#1 Sieve of Eratosthenes - O(n log log n) time to generate primes up to n

vector<int> generatePrimes(int n){
	bool prime[n+1];
	memset(prime, true, sizeof(prime));

	int m = sqrt(n);
	for(int i = 2; i <= m; ++i){
		if(prime[i]){
			// mark all multiples of i as not prime
			for(int j = i+i; j <= n; j += i){
				prime[j] = false;
			}
		}
	}

	vector<int> primes;
	for(int i = 2; i <= n; ++i)
		if(prime[i]) primes.push_back(i);

	return primes;
}
